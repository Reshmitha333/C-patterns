#include <stdio.h>

int main()
{
    int i,j,n,k,m;
    printf("Enter n value");
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        k=i;
        m=i+1;
        for(j=1;j<=i;j++)
        {
            printf("%d",k);
            k--;
        }
        for(j=i;j<=n;j++)
        {
            printf("%d",m);
            m++;
        }
        printf("\n");
    }

    return 0;
}
